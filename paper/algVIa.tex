  \SetKwFunction{GoodPartitionJStrip}{\textsc{Good-Partition-From-J-Strip-System}}
	\Indm\nonl\GoodPartitionJStrip{$G, J, (S, N), M$}\\
  \KwData{$G$ -- square-free, Berge graph \newline
